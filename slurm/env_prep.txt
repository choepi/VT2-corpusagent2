# find available gpu Nodes 
sinfo -h -p gpu,gpu_ia,gpu_top -N -o '%N|%P' \
| while IFS='|' read -r node part; do
  scontrol show node "$node" | awk -v node="$node" -v part="$part" '
    /Gres=/ {
      # find the field that starts with Gres=
      for (i = 1; i <= NF; i++) {
        if ($i ~ /^Gres=/) {
          gres = $i
          sub(/^Gres=/, "", gres)
        }
      }
    }
    /AllocTRES=/ {
      # capture full AllocTRES=... string
      for (i = 1; i <= NF; i++) {
        if ($i ~ /^AllocTRES=/) {
          alloc = $i
        }
      }
    }
    END {
      # ---- parse GPU model + total count ----
      total = 0
      model_str = ""

      n = split(gres, garr, ",")
      for (i = 1; i <= n; i++) {
        if (garr[i] ~ /^gpu:/) {
          # e.g. gpu:v100sxm:8 or gpu:a100pcie:3
          split(garr[i], b, ":")
          # b[2] = model, b[3] = count
          if (model_str == "")
            model_str = b[2]
          else
            model_str = model_str "," b[2]

          if (b[3] != "")
            total += b[3] + 0
        }
      }

      # ---- parse allocated GPUs from AllocTRES= ----
      alloc_gpu = 0
      sub(/^AllocTRES=/, "", alloc)
      m = split(alloc, aarr, ",")
      for (i = 1; i <= m; i++) {
        # look for gres/gpu=X
        if (aarr[i] ~ /^gres\/gpu=/) {
          split(aarr[i], c, "=")
          alloc_gpu += c[2] + 0
        }
      }

      free = total - alloc_gpu

      printf "%-12s %-10s %-20s total=%2d alloc=%2d free=%2d\n",
             node, part, model_str, total, alloc_gpu, free
    }'
done
